t->setDiagnosis(s_temp);
patient >> i_temp;
t->setDays(i_temp);
patient >> s_temp;
t->setDepartment(s_temp);
patient >> s_temp;
t->setDate(s_temp);
if(t->getDepartment() == _name)
_list.add(t);
}
if(_list.size()>=0)
_list.popHead();
patient.close();

in_hospital >> i_temp;

in_hospital.close();
}
